public  static int[] swap(int [] values)
	{
			int i=0;
			int j=values.length/2;
			while(i<values.length/2)
			{
				int temp=values[i];
				values[i]=values[j];
				values[j]=temp;
				i++;
				j++;
			}
			return values;
	}
